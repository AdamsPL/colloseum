PROJECT(fightclub)

SET(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -ggdb -Wall -pedantic ")

MACRO(FIND_CMAKE_DIRS_RECURSIVE result)
	FILE(GLOB_RECURSE cmakelists CMakeLists.txt)
	FOREACH(cmake ${cmakelists})
		GET_FILENAME_COMPONENT(dirname ${cmake} PATH)
		IF (NOT ${dirname} STREQUAL ${CMAKE_CURRENT_LIST_DIR})
			MESSAGE(STATUS "Found cmake dir: " ${dirname})
			ADD_SUBDIRECTORY(${dirname})
		ENDIF()
	ENDFOREACH()
ENDMACRO()
